   538→      }

   540→      toast({ title: 'خطأ', description: 'فشل في الاتصال بالخادم', variant: 'destructive' })

   614→              </div>

   616→                <h1 className="text-lg font-bold text-[#D4AF37]">لوحة الكاشير</h1>

   621→              <Button variant="ghost" onClick={onLogout} className="gap-2 text-muted-foreground hover:text-red-400">
   622→                <X className="h-4 w-4" />خروج

   642→          <motion.div initial={{ y: -80 }} animate={{ y: 0 }} exit={{ y: -80 }}

   675→              { value: 'active', icon: ShoppingBag, label: 'الطلبات النشطة', shortLabel: 'النشطة', count: activeOrders.length, highlight: readyOrders.length > 0 },

   677→              { value: 'expenses', icon: TrendingDown, label: 'المصروفات', shortLabel: 'المصروفات', count: expenses.length },

   700→                    <span className={`flex h-5 w-5 items-center justify-center rounded-full text-[10px] text-white font-bold ${tab.highlight ? 'bg-green-500' : 'bg-[#D4AF37]/70'}`}>

   710→        {/* Active Orders */}

   750→                <Receipt className="mx-auto mb-4 h-16 w-16 text-muted-foreground/20" />
